func traverseAndPrint(head) {
    var currentNode = head
    while (currentNode != null) {
        printf(" %d ->", currentNode.val)
        currentNode = currentNode.next
    }
    print("null")
}

func findLowestValue(head){
  var minValue = head.val
  var currentNode = head.next
  while (currentNode != null) {
    if (currentNode.val < minValue) {
      minValue = currentNode.val
    }
    currentNode = currentNode.next
  }

  return minValue
}

func deleteSpecificNode(head, nodeToDelete) {
    if (head == nodeToDelete){
        return head.next
    }

    var currentNode = head
    while (currentNode.next != null and currentNode.next != nodeToDelete) {
        currentNode = currentNode.next
    }

    if (currentNode.next == null) {
        return head
    }

    currentNode.next = currentNode.next.next

    return head

}
